### 2. **Python集成脚本**

```python
#!/usr/bin/env python3
"""
微信公众号热榜抓取器 - 基于今日热榜API
"""

import requests
import json
import time
import logging
from typing import List, Dict

class WechatHotFetcher:
    """微信公众号热榜抓取器"""
    
    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = "https://api.tophubdata.com"
        self.session = requests.Session()
        self.session.headers.update({
            "Authorization": api_key,
            "User-Agent": "Mozilla/5.0"
        })
    
    def fetch_wechat_hot(self, limit: int = 10) -> List[Dict]:
        """抓取微信公众号热榜"""
        try:
            # 微信公众号热榜hashid
            hashid = "WnBe01o371"
            
            response = self.session.get(
                f"{self.base_url}/nodes/{hashid}",
                timeout=10
            )
            
            if response.status_code == 200:
                data = response.json()
                
                if not data.get("error"):
                    items = data["data"].get("items", [])
                    
                    # 转换为标准格式
                    topics = []
                    for i, item in enumerate(items[:limit]):
                        topics.append({
                            "选题标题": item.get("title", ""),
                            "来源平台": "微信公众号",
                            "热度指数": self._parse_hotness(item.get("extra", "0")),
                            "链接": {
                                "link": item.get("url", ""),
                                "text": "查看文章",
                                "type": "url"
                            },
                            "发布时间": int(time.time() * 1000),
                            "内容类型": self._detect_category(item.get("title", "")),
                            "状态": "待分析",
                            "备注": "微信公众号热榜（今日热榜API）",
                            "关键词": self._extract_keywords(item.get("title", ""))
                        })
                    
                    return topics
        
        except Exception as e:
            logging.error(f"抓取微信公众号热榜失败: {e}")
        
        return []
    
    def _parse_hotness(self, extra: str) -> int:
        """解析热度值"""
        try:
            # 示例: "10.0万" -> 100000
            if "万" in extra:
                num = float(extra.replace("万", "").strip())
                return int(num * 10000)
            else:
                return int(extra)
        except:
            return 0
    
    def _detect_category(self, title: str) -> str:
        """检测内容类型"""
        categories = {
            "科技": ["科技", "AI", "人工智能", "互联网", "数码"],
            "财经": ["财经", "经济", "股票", "投资", "金融"],
            "生活": ["生活", "健康", "养生", "习惯", "日常"],
            "娱乐": ["娱乐", "明星", "影视", "音乐", "综艺"],
            "时事": ["新闻", "政策", "通知", "公告", "政府"],
            "教育": ["教育", "学习", "学校", "考试", "培训"]
        }
        
        for category, keywords in categories.items():
            for keyword in keywords:
                if keyword in title:
                    return category
        
        return "综合"
    
    def _extract_keywords(self, title: str) -> List[str]:
        """提取关键词"""
        keywords = ["微信公众号", "热榜"]
        
        # 添加标题中的关键词
        words = title.split()
        for word in words[:3]:
            if len(word) > 1:
                keywords.append(word)
        
        return keywords[:5]

# 使用示例
def main():
    # 配置API密钥
    API_KEY = "YOUR_ACCESS_KEY_HERE"
    
    # 创建抓取器
    fetcher = WechatHotFetcher(API_KEY)
    
    # 抓取热榜
    topics = fetcher.fetch_wechat_hot(limit=5)
    
    print(f"抓取到 {len(topics)} 条微信公众号热榜")
    
    for i, topic in enumerate(topics):
        print(f"{i+1}. {topic['选题标题'][:30]}...")
        print(f"   热度: {topic['热度指数']}")
        print(f"   链接: {topic['链接']['link'][:50]}...")
        print()

if __name__ == "__main__":
    main()
```
